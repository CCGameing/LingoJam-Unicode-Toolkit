How it works:

Unicode is how characters are rendered in your computer/browser! They are used in Hexidecimal (Unicode) Values, but are stored in Decimal (Code Point) Values. Most Unicode Characters use 16 bits, but some can use 32 bits. This complicates things when storing them in strings.

Hints:

  • Use '+' to add two 16 bit characters into a 32 bit character
  • Open the Console to see the characters being generated

Forward Translation

function forward(input) {

    var list = input.split(&quot; &quot;) // Split the string from the Hex input box into an array

    for (var i = 0; i &lt; list.length; i++) {

        var value = parseInt(list[i], 16); // At this point in the array, Hex to Dec
        var str = String.fromCodePoint(value); // Create a character from the CP value
        list[i] = str // Replace this point with new Character

    }

return list.join(''); // Return a string containing the new characters

}

Backward Translation

function backward(input) {

    var list = input.split('') // Split the string into an array

    for (var i = 0; i &lt; list.length; i++) {

        if (input[i]=='+'&amp;&amp;input[i+1]!=' ') { 
        // Basically '+ü§ì' == true and '+ ' == false

            // 'ü§ì' is a 32 bit character, which means if it is used in a string, it gets split into two 16 bit characters.

            var half1 = list[i+1]; 
            var half2 = list[i+2];
            //We can fix this by adding the two 16 bit characters together

            var value = (half1 + half2).codePointAt(0); 
            // and getting the CP value of the whole charater

            var str = value.toString(16); 
            // Dec to Hex

            list[i] = str;
            list[i+1] = '';
            list[i+2] = '';
            i += 2
            // Because we added the next two characters together, we can skip them afterward.

        } else {

            var value = list[i].codePointAt(0); // Get the CP value
            var str = value.toString(16); // Make it a Unicode value
            list[i] = str;

        }

    }

    return list.join(&quot; &quot;);

}
